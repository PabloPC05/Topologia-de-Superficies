\section{Conexidad}

\subsection{Espacios conexos}

\begin{definiciĂ³n}[SeparaciĂ³n]
    Una \textbf{separaciĂ³n} $U|V$ de un espacio topolĂ³gico $X$ es un par de abiertos disjuntos $U$, $V$, tales que $X = U \cup V$. Una separaciĂ³n $\emptyset|X$ (o $X|\emptyset$) se denomina \textbf{separaciĂ³n trivial}.
\end{definiciĂ³n}
\begin{definiciĂ³n}[Conexo]
    Se dice que un espacio topolĂ³gico $X$ es \textbf{conexo} cuando no admite ninguna separaciĂ³n distinta de la trivial. Si $A \subset X$, se dice que $A$ es un \textbf{conjunto conexo} cuando es conexo como subespacio.
\end{definiciĂ³n}
\begin{proposiciĂ³n}[CaracterizaciĂ³n de la conexidad]
    La conexidad de un espacio $X$ es equivalente a cada una de las siguientes propiedades:
    \begin{enumerate}[label=\alph*)]
        \item Los Ăºnicos subconjuntos de $X$ que son, simultĂ¡neamente, abiertos y cerrados son $\emptyset$ y $X$.
        \item Toda aplicaciĂ³n continua de $X$ en $\{0, 1\}$ con la discreta ($\tau = \{\emptyset, \{0, 1\}, \{1\}, \{0\}\}$) es constante. 
        \item Para todo par de puntos de $X$ existe un subconjunto conexo de $X$ que los contiene.
    \end{enumerate}
\end{proposiciĂ³n}
\begin{proof}
    \begin{enumerate}
        \item $\rightarrow$: Supongamos que existe otro conjunto $A$ que es abierto y cerrado. Entonces $A|\overline{A}$ es una separaciĂ³n de $X$ distinta de la trivial, por lo tanto $X$ no es conexo. \\ 
        $\leftarrow$: Supongamos que $X$ no es conexo, es decir, existe una separaciĂ³n $U|V$ de $X$ distinta de la trivial. Entonces $U$ y $V$ son abiertos y cerrados distintos de $\emptyset$ y $X$.
        \item $\rightarrow$: Supongamos que $X$ es conexo, pero existe $f: X \to \{0, 1\}$ continua pero no constante. Eso significa que podemos definir dos conjuntos: 
    $$U = f^{-1}(\{0\}) \text{ y } V = f^{-1}(\{1\}).$$
    Como ambos $\{0\}$ y $\{1\}$ son abiertos y $f$ es continua, entonce $u$ y $V$ son abiertos con $U \cap V = \emptyset$ y $U \cup V = X$, es decir, $U|V$ es una separaciĂ³n de $X$ distinta de la trivial, lo que contradice la hipĂ³tesis de que $X$ es conexo. \\
    $\leftarrow$: Supongamos que toda aplicaciĂ³n continua de $X$ en $\{0, 1\}$ es constante, pero $X$ no es conexo. Entonces existe una separaciĂ³n $U|V$ de $X$ distinta de la trivial. Definamos $f: X \to \{0, 1\}$ por $$f(x) = \begin{cases} 0 & \text{si } x \in U \\ 1 & \text{si } x \in V \end{cases}$$
    La aplicaciĂ³n $f$ es continua, pero no constante, lo que contradice la hipĂ³tesis.
    \item $\rightarrow$: Si el espacio $X$ ya es conexo, dados dos puntos cualesquiera $x, y \in X$, el conjunto $X$ es un subconjunto conexo de $X$ que los contiene. \\
    $\leftarrow$: Supongamos que para todo par de puntos $x, y \in X$ existe un subconjunto conexo $C \subseteq X$ con $x, y \in C$, pero que $X$ no es conexo. Entonces existe una separaciĂ³n no trivial $U|V$ de $X$. Como $U \neq \emptyset$ y $V \neq \emptyset$, podemos elegir $x \in U$ e $y \in V$. Por hipĂ³tesis, existe $C \subseteq X$ conexo con $x, y \in C$. Consideremos $U_C = C \cap U$ y $V_C = C \cap V$: son abiertos en $C$ (por la topologĂ­a del subespacio), disjuntos, $U_C \cup V_C = C \cap (U \cup V) = C$, y ninguno es vacĂ­o pues $x \in U_C$ e $y \in V_C$. Luego $U_C|V_C$ es una separaciĂ³n no trivial de $C$, lo que contradice que $C$ es conexo.
    \end{enumerate}
\end{proof}
\ejemplo{Algunos ejemplos de espacios conexos son los triviales, discretos, Sierpinski, Kolmogorov, $\R$, $\Q$.}
\begin{proposiciĂ³n}
    Sea $\tau$ y $\tau'$ dos topologĂ­as en un conjunto $X$ con $\tau \subset \tau'$. Si $(X, \tau')$ es conexo, entonces $(X, \tau)$ es conexo. \\
    Es decir, la conexidad se conserva al debilitar la topologĂ­a.
\end{proposiciĂ³n}
\begin{proof}
    Sean $\tau, \tau^{\prime}$ dos topologĂ­as sobre un mismo conjunto tales que $\tau \subset \tau^{\prime}$, pero supongamos que $(X, \tau^{\prime}$ es conexo, pero $(X, \tau)$ no lo es. Entonces sabemos que existe una separaciĂ³n $U|V$ de $(X, \tau)$ distinta de la trivial. No obstante, dado que $\tau \subset \tau^{\prime}$, tenemos que $U, V \in \tau^{\prime}$, por lo que $U|V$ es una separaciĂ³n de $(X, \tau^{\prime})$ distinta de la trivial, lo que contradice la hipĂ³tesis de que $(X, \tau^{\prime})$ es conexo.
\end{proof}
\begin{proposiciĂ³n}
    Si $U|V$ es una separaciĂ³n de $X$ y $A \subset X$ es conexo, entonces $A \subset U$ o $A \subset V$.
\end{proposiciĂ³n}
\begin{proof}
    Sean $U, V \in \tau$ tales que $U|V$ es una separaciĂ³n de $X$ y sea $A \subset X$ conexo, pero supongamos que $A \cap V \neq \emptyset$ y $A \cap U \neq \emptyset$. Por tanto podemos definir 
    $$U_A = A \cap U \text{ y } V_A = A \cap V.$$
    Ambos $U_A$ y $V_A$ son abiertos en $A$, por ser una topologĂ­a relativa. Y dado que son una separaciĂ³n en $X$, se ha de cumplir que $U_A \cup V_A = A \cap (U \cup V) = A$. No obstante, tenemos que $A$ es conexo, por lo que se ha de daar que $U_A = A$ y $V_A = \emptyset$ (o viceversa), lo que contradice la hipĂ³tesis de que $A \subset U$ y $A \subset V$.
\end{proof}
\begin{proposiciĂ³n}
    Si $A \subset X$ es un conjunto conexo y $A \subset B \subset Cl{A}$, entonces $B$ es conexo.
\end{proposiciĂ³n}
\begin{proof}
    Supongamos que bajo las condiciones propuestas, $B$ es disconexo, entonces existe una separaciĂ³n $U|V$ de $B$ distinta de la trivial. Por tanto, $U \cap A$ y $V \cap A$ son abiertos en $A$, disjuntos, $A \cap (U \cup V) = A$, y ninguno es vacĂ­o, lo que contradice que $A$ es conexo.
\end{proof}
\begin{proposiciĂ³n}
    Si $\{A_\mu\}_{\mu \in M}$ es una familia de subconjuntos conexos de un espacio topolĂ³gico $X$ y $\bigcap_{\mu \in M} A_\mu \neq \emptyset$, entonces $\bigcup_{\mu \in M} A_\mu$ es conexo.
\end{proposiciĂ³n}
\begin{proof}
    Bajo las condiciones propuestas, supongamos que $\bigcup_{\mu \in M} A_\mu$ es disconexo, entonces existe una separaciĂ³n $U|V$ de $\bigcup_{\mu \in M} A_\mu$ distinta de la trivial. AdemĂ¡s, dado que $\bigcap_{\mu \in M} A_\mu \neq \emptyset$, existe $x_0 \in X$ tal que $x_0 \in A_\mu$ para cada $\mu \in M$. Por tanto, $x_0 \in U$ o $x_0 \in V$, supongamos que $x_0 \in U$, y ademĂ¡s que $\forall \mu \in M$, $x_0 \in A_\mu$. \\
    TambiĂ©n tenemos que $\forall \mu \in M$, $A_{\mu} \subset \bigcup_{\mu \in M} A_\mu$, por lo que tenemos que 
    $$A_{\mu} = (A_{\mu} \cap V) \cup (A_{\mu} \cap U)$$
    Pero como $A_{\mu} \cap U$ y $A_{\mu} \cap V$ son abiertos en $A_{\mu}$, disjuntos, y $A_{\mu}$ es conexo, se ha de cumplir que $A_{\mu} \subset U$ (dado que $x_0 \in U$). Dado que esto se cumple para cada $\mu \in M$, se ha de cumplir que $\bigcup_{\mu \in M} A_\mu \subset U$, lo que contradice la hipĂ³tesis de que $U|V$ es una separaciĂ³n de $\bigcup_{\mu \in M} A_\mu$ distinta de la trivial.
\end{proof}
\begin{proposiciĂ³n}
    Si $A, B \subset X$ son conexos y $Cl(A) \cap B \neq \emptyset$, entonces $A \cup B$ es conexo.
\end{proposiciĂ³n}
\begin{proof}
    Bajo las condiciones propuestas supongamos que $A \cup B$ es disconexo, entonces existe una separaciĂ³n $U|V$ de $A \cup B$ distinta de la trivial. Esto implica que 
    $$\begin{cases} A = (A \cap U) \cup (A \cap V) \\ B = (B \cap U) \cup (B \cap V) \end{cases}$$
    Pero como $A \cap U$ y $A \cap V$ son abiertos en $A$, disjuntos, y $A$ es conexo, se ha de cumplir que $A \subset U$ o $A \subset V$, supongamos que $A \subset U$. Asimismo, debemos suponer que $B \subset V$ (ya que en caso contrario tendriamos que $B \subset U$ y habrĂ­amos acabado $A \cup B \subset U$). Ahora, dado que $Cl(A) \cap B \neq \emptyset$, existe $x_0 \in X$ tal que $x_0 \in Cl(A)$ y $x_0 \in B$ y por tanto $x_0 \in B \subset V$. Pero tenĂ­amos que $U, V \in \tau_{A \cup B}$, por lo que $\exists W \in \tau$ tal que 
    $$V = W \cap (A \cup B)$$
    Entonces tenemos que $x_0 \in W$, por lo que, por la definiciĂ³n de clausura tenemos que $W \cap A \neq \emptyset \iff \exists y \in W \cap A$. Y esto signfica que $y \in A \cup B$ por lo que $y \in W \cap (A \cup B) \implies y \in V$. Pero habĂ­amos supuesto que $A \subset U$, por lo que $y \in A \subset U$, lo que contradice que $U$ y $V$ son disjuntos.
\end{proof}

\begin{proposiciĂ³n}[Imagen continua de un conexo]
    Si $X$ es conexo y $f \colon X \to Y$ es sobreyectiva y continua, entonces $Y$ es conexo.
\end{proposiciĂ³n}
\begin{proof}
     Supongamos que $X$ es conexo, pero $Y$ no lo es, entonces quiere decir que existe una separaciĂ³n $U|V$ distinta de la trivial. Dado que $f$ es continua, $f^{-1}(U)$ y $f^{-1}(V)$ son abiertos en $X$. AdemĂ¡s, dado que $f$ es sobreyectiva, se ha de cumplir que $f^{-1}(U) \neq \emptyset $ y $\cup f^{-1}(V) \neq \emptyset$. Por lo que tenemos que $f^{-1}(U) \cup f^{-1}(V) = f^{-1}(U \cup V) = f^{-1}(Y) = X$. Por Ăºltimo, dado que $U$ y $V$ son disjuntos, se ha de cumplir que $f^{-1}(U) \cap f^{-1}(V) = f^{-1}(U \cap V) = f^{-1}(\emptyset) = \emptyset$. Por tanto, $f^{-1}(U)|f^{-1}(V)$ es una separaciĂ³n de $X$ distinta de la trivial, lo que contradice la hipĂ³tesis de que $X$ es conexo.
\end{proof}
\begin{corolario}
    La conexidad es una propiedad topolĂ³gica. Es decir se conserva bajo homeomorfismo ($f$ es continua, biyectiva y $f^{-1}$ es continua).
\end{corolario}
\begin{proof}
    Por la proposiciĂ³n anterior, si $X$ es conexo y $f \colon X \to Y$ es continua y sobreyectiva, entonces $Y$ es conexo. Dado que $f$ es un homeomorfismo, $f^{-1} \colon Y \to X$ tambiĂ©n es continua y sobreyectiva, por lo que si $Y$ es conexo, entonces $X$ es conexo. 
\end{proof}
\begin{corolario}
    Todo cociente de un conexo es conexo.
\end{corolario}
\begin{proof}
    Sea $X$ un espacio conexo y $X/\sim$ un cociente de $X$. Entonces, por definiciĂ³n de cociente, existe una aplicaciĂ³n continua y sobreyectiva $f \to X/\sim$ tal que $f(x) = [x]$ para cada $x \in X$. Por la proposiciĂ³n anterior, $X/\sim$ es conexo.
\end{proof}
\begin{teorema}[Teorema de Bolzano o teorema del valor intermedio]
    Sean $X$ conexo, $f \colon X \to \R$ continua y $x, y \in X$ tales que $f(x) < f(y)$. Entonces, para cada $t \in \R$ con $f(x) < t < f(y)$, existe $z \in X$ tal que $f(z) = t$.
\end{teorema}
\begin{proof}
    Supongamos, por reducciĂ³n al absurdo, que existe un $t \in \mathbb{R}$ con $f(x) < t < f(y)$ tal que no existe ningĂºn $z \in X$ que cumpla $f(z) = t$. Esto significa que $t \notin f(X)$.

    Definamos los intervalos $U = (-\infty, t)$ y $V = (t, \infty)$. Es claro que $U$ y $V$ son conjuntos abiertos en la topologĂ­a usual de $\mathbb{R}$. 
    
    Como $f$ es una funciĂ³n continua, las preimĂ¡genes $f^{-1}(U)$ y $f^{-1}(V)$ son conjuntos abiertos en $X$. Veamos que estos dos conjuntos forman una separaciĂ³n de $X$:
    
    \begin{itemize}
        \item \textbf{Son no vacĂ­os:} Como $f(x) < t$, se tiene que $f(x) \in U$, luego $x \in f^{-1}(U)$. AnĂ¡logamente, como $f(y) > t$, se tiene que $f(y) \in V$, luego $y \in f^{-1}(V)$. Por tanto, $f^{-1}(U) \neq \emptyset$ y $f^{-1}(V) \neq \emptyset$.
        \item \textbf{Son disjuntos:} Como $U \cap V = \emptyset$, entonces $f^{-1}(U) \cap f^{-1}(V) = f^{-1}(U \cap V) = f^{-1}(\emptyset) = \emptyset$.
        \item \textbf{Cubren a $X$:} Dado que por nuestra suposiciĂ³n original no hay ningĂºn elemento en $X$ cuya imagen sea $t$, para cualquier $w \in X$ se debe cumplir obligatoriamente que $f(w) < t$ o $f(w) > t$. Es decir, $f(w) \in U \cup V$. Por lo tanto, todo $w \in X$ pertenece a $f^{-1}(U)$ o a $f^{-1}(V)$, lo que implica que $X = f^{-1}(U) \cup f^{-1}(V)$.
    \end{itemize}
    
    Hemos encontrado que $f^{-1}(U) | f^{-1}(V)$ es una separaciĂ³n de $X$ formada por dos abiertos disjuntos y no vacĂ­os. Esto contradice directamente la hipĂ³tesis de que $X$ es conexo. 
    
    La contradicciĂ³n proviene de suponer que no existĂ­a $z \in X$ con $f(z) = t$. Por lo tanto, concluimos que tal $z$ debe existir.
\end{proof}
\begin{definiciĂ³n}[TopologĂ­a Producto]
    Dados dos espacios topolĂ³gicos $X$ e $Y$, la \textbf{topologĂ­a producto} en $X \times Y$ es la topologĂ­a generada por las bases de la forma $U \times V$, donde $U$ es abierto en $X$ y $V$ es abierto en $Y$.
\end{definiciĂ³n}
\begin{proposiciĂ³n}[Producto de conexos]
    Un producto $X \times Y$ es conexo si, y solo si, $X$ e $Y$ son conexos.
\end{proposiciĂ³n}
\begin{proof}
    $\rightarrow$: Supongamos que $X \times Y$ es conexo, pero $X$ no lo es, entonces existe una separaciĂ³n $U|V$ de $X$ distinta de la trivial, entonces $$U \times Y | V \times Y$$ es una separaciĂ³n de $X \times Y$ distinta de la trivial, lo que contradice la hipĂ³tesis.\\
    $\leftarrow$: Supongamos que $X$ e $Y$ son conexos, entonces, dado que sabemos que es una propiedad topolĂ³gica sabemos que, dados $x_0 \in X$ y $y_0 \in Y$ podemos afirmar que $X \times \{y_0\}$ y $\{x_0\} \times Y$ son conexos. AsĂ­, podemos definir la familia de conjuntos: 
    $$T_x = (X \times \{y_0\}) \cup (\{x\} \times Y)$$
    El cual es un conjunto conexo dado que la interesecciĂ³n de ambos conjuntos no es nula. AsĂ­ formamos la famlia de conjuntos $\{T_x\}_{x \in X}$, la cual es una familia de conjuntos conexos con intersecciĂ³n no vacĂ­a, por lo que su uniĂ³n es conexa. Pero la uniĂ³n de esta familia de conjuntos es $X \times Y$, por lo que $X \times Y$ es conexo.
\end{proof}

\subsection{Componentes conexas}

\begin{definiciĂ³n}[Componentes conexas]
    Se llaman \textbf{componentes conexas} de un espacio topolĂ³gico $X$ a los subconjuntos de $X$ que son conexos maximales con respecto a la inclusiĂ³n. Es decir, $C \subset X$ es una componente conexa si $C$ es conexo y para todo $Y$ con $C \subsetneq Y \subset X$, $Y$ no es conexo.
\end{definiciĂ³n}
\ejemplo{Algunos ejemplos: espacios conexos, discretos, espacios \textbf{totalmente disconexos}, $\R \setminus \{0\}$, $\R \setminus \Z$, $\R \setminus \Q$.}
\begin{proposiciĂ³n}
    Todo subconjunto conexo no vacĂ­o de un espacio $X$ estĂ¡ contenido en una Ăºnica componente conexa de $X$.
\end{proposiciĂ³n}
\begin{proof}
    Sea $A \subset X$ un conjunto conexo no vacĂ­o. Tomemos $x \in A$. Por la definiciĂ³n de componente conexa, existe una componente conexa $C(x)$ que contiene al punto. Dado que $A$ es un conjunto conexo y contiene al punto, $C(X)$ es la uniĂ³n de todos los conjuntos conexos que contienen a $x$, por lo que se deduce que
    $$A \subset C(x)$$
    Esto prueba que $A$ estrĂ¡ contenido en al menos una componente conexa. \\
    Ahora veamos la unicidad: Supongamos que $A$ estĂ¡ contenido en dos componentes conexas, $C_1$ y $C_2$. Como $A$ no es vacĂ­o, existe $a \in A$ tal que $a \in C_1$ y $a \in C_2$. Por lo tanto, $C_1 \cap C_2 \neq \emptyset$ y dado que son maximales, al compartir un punto han de ser iguales, es decir, $C_1 = C_2$.
\end{proof}
\begin{corolario}
    La familia de las componentes conexas de un espacio $X$ es una descomposiciĂ³n de $X$.
\end{corolario}
\begin{proof}
    Debemos probar que la familia de componentes conexas $\{C_\mu\}_{\mu}$ cumple las tres condiciones de una descomposiciĂ³n:
    \begin{enumerate}
        \item \textbf{Recubrimiento:} Sea $x \in X$. El conjunto $\{x\}$ es conexo (trivialmente) y no vacĂ­o, por lo que, por la proposiciĂ³n anterior, estĂ¡ contenido en una Ăºnica componente conexa $C(x)$. En particular, $x \in C(x)$, luego todo punto de $X$ pertenece a alguna componente conexa, es decir, $X = \bigcup_{\mu} C_\mu$.
        \item \textbf{No vacuidad:} Cada componente conexa es, por definiciĂ³n, un subconjunto conexo maximal, y por la proposiciĂ³n anterior contiene al menos un punto, luego es no vacĂ­a.
        \item \textbf{DisjunciĂ³n:} Supongamos que dos componentes conexas $C_1$ y $C_2$ cumplen $C_1 \cap C_2 \neq \emptyset$. Entonces existe $x \in C_1 \cap C_2$. Como $\{x\}$ es conexo y no vacĂ­o, por la unicidad demostrada en la proposiciĂ³n anterior, la componente conexa que lo contiene es Ăºnica. Dado que tanto $C_1$ como $C_2$ contienen a $x$, se concluye que $C_1 = C_2$.
    \end{enumerate}
    Por tanto, la familia de componentes conexas es una descomposiciĂ³n de $X$.
\end{proof}
\begin{definiciĂ³n}[Familia de componentes conexas]
    Sea $X$ un espacio topolĂ³gico. La \textbf{familia de componentes conexas} de $X$ es la descomposiciĂ³n $\Gamma(X)$ de $X$ formada por las componentes conexas de $X$.
\end{definiciĂ³n}
\begin{definiciĂ³n}[FunciĂ³n $\Gamma$]
    Sea $f: X \to Y$ una funciĂ³n continua, definimos la aplicaciĂ³n inducida $$\Gamma(f): \Gamma(X) \to \Gamma(Y)$$
    Tal que para toda componente conexa $C \in \Gamma(X)$, sabemos que $f(C)$ es tambiĂ©n un subconjunto conexo de $Y$, por lo que tambiĂ©n debe estar contenido en una Ăºnica componente conexa de $Y$. Entonces se define precisamente esa Ăºnica componente conexa como $\Gamma(f)(C)$.
\end{definiciĂ³n}
\begin{proposiciĂ³n}
    Si $f \colon X \to Y$ es un homeomorfismo, entonces $\Gamma(f) \colon \Gamma(X) \to \Gamma(Y)$ es biyectiva y, para cada $C \in \Gamma(X)$, $\Gamma(f)(C)$ es homeomorfa a $C$.
\end{proposiciĂ³n}
\begin{proof}
    Veamos la demostraciĂ³n por puntos: 
    \begin{itemize}
        \item $\Gamma(f)$ es una biyecciĂ³n: Como $f$ es un homeomorfismo su inversa es continua y usando la propiedad de las aplicaciones inducida dada por
        $$\Gamma(g \circ f) = \Gamma(g) \circ \Gamma(f), \quad \Gamma(\operatorname{Id}_X) = \operatorname{Id}_{\Gamma(X)}$$
        Entonces, aplicando esto a $f$ y a $f^{-1}$:
        $$\Gamma(f^{-1}) \circ \Gamma(f) = \Gamma(f^{-1} \circ f) = \Gamma(\operatorname{Id}_X) = \operatorname{Id}_{\Gamma(X)}$$
        $$\Gamma(f) \circ \Gamma(f^{-1}) = \Gamma(f \circ f^{-1}) = \Gamma(\operatorname{Id}_Y) = \operatorname{Id}_{\Gamma(Y)}$$
        Como $\Gamma(f)$ tiene una inversa por ambos lados, concluimos que $\Gamma(f)$ es una biyecciĂ³n.
        \item $C$ es homeomorfo a $\Gamma(f)(C)$: Sea $C \in \Gamma(X)$ y sea $K = \Gamma(f)(C)$. Por definiciĂ³n, $K$ es la Ăºnica componente conexa de $Y$ que contiene a $f(C)$, es decir, $f(C) \subset K$. AnĂ¡logamente, como $\Gamma(f)$ es biyectiva con inversa $\Gamma(f^{-1})$, se tiene que $C$ es la Ăºnica componente conexa de $X$ que contiene a $f^{-1}(K)$, es decir, $f^{-1}(K) \subset C$. Esto implica que $K \subset f(C)$, y por tanto $f(C) = K$. AsĂ­, la restricciĂ³n $f|_C \colon C \to K$ es sobreyectiva. Como $f$ es inyectiva, $f|_C$ tambiĂ©n lo es. AdemĂ¡s, $f|_C$ es continua por ser restricciĂ³n de una funciĂ³n continua. Finalmente, su inversa $(f|_C)^{-1} = f^{-1}|_K \colon K \to C$ es tambiĂ©n continua por ser restricciĂ³n de $f^{-1}$, que es continua por ser $f$ un homeomorfismo. Por tanto, $f|_C \colon C \to K$ es un homeomorfismo.
    \end{itemize}
\end{proof}
\begin{observaciĂ³n}
    El cardinal de $\Gamma(X)$ es un \textbf{invariante topolĂ³gico} y puede usarse, por ejemplo, para comprobar que $\R \not\approx \R^n$ si $n > 1$, $[0, +\infty) \not\approx \R$, $[0, 1] \not\approx \mathbb{S}^1$ o $\mathbb{S}^1 \not\approx \mathbb{S}^n$ si $n > 1$.
\end{observaciĂ³n}
\begin{proposiciĂ³n}
    Las componentes conexas de un espacio $X$ son cerradas en $X$.
\end{proposiciĂ³n}
\begin{proof}
    Sea $C$ una componente conexa de $X$. Sabemos que un conjunto es cerrado si y solo si $A = Cl(A)$, por lo que basta con probar que $C = Cl(C)$. Sabemos por una proposiciĂ³n anterior que si $A$ es conexo, entonces cualquier conjunto $B$ con $A \subset B \subset Cl(A)$ es conexo. 
    En particular, si tomamos $A = C$ y $B = Cl(C)$, entonces el teorema nos dice que $Cl(C)$ es conexo. Pero por la definiciĂ³n de maximalidad de $C$, tenemos que no puede exitir ningĂºn conjunto conexo que contenga a $C$ y sea mĂ¡s grande que Ă©l. Por tanto, tenemos que $Cl(C)$ es conexo y que $C \subset Cl(C)$, por lo que se ha de dar necesariamente que $C = Cl(C)$, es decir, $C$ es cerrado.
\end{proof}
\begin{proposiciĂ³n}
    Si $A \subset X$ es no vacĂ­o, conexo, abierto y cerrado, entonces $A$ es una componente conexa de $X$.
\end{proposiciĂ³n}
\begin{proof}

Por lo que en la topologĂ­a relativa de $C$, los conjuntos $A$ y $C \cap A^c$ son tambiĂ©n abiertos (por ser intersecciones de $C$ con abiertos de $X$). Es decir, hemos conseguido escribir $C$ como la uniĂ§on de dos conjuntos abiertos disjuntos y no vacĂ­os. sto implicarĂ­a que $C$ es disconexo, lo que contradice la hipĂ³tesis de que $C$ es una componente conexa. \end{proof}

\subsection{Conexidad por caminos}

\begin{observaciĂ³n}
    Denotaremos por $\mathbb{I}$ el intervalo $[0, 1]$ con la topologĂ­a usual.
\end{observaciĂ³n}
\begin{definiciĂ³n}[Camino]
    Un \textbf{camino} en un espacio topolĂ³gico $X$ es una aplicaciĂ³n continua de $\mathbb{I}$ en $X$. Si $\varphi \colon \mathbb{I} \to X$ es un camino en $X$, los puntos $\varphi(0)$ y $\varphi(1)$ se llaman \textbf{extremos} del camino (\textbf{punto inicial} y \textbf{punto final}) y se dice que $\varphi$ es un camino de $\varphi(0)$ a $\varphi(1)$.
\end{definiciĂ³n}
\ejemplo{Algunos ejemplos de caminos: constantes; no triviales; no finitos; segmentos y poligonales en $\R^n$.}
\begin{observaciĂ³n}[Camino inverso y yuxtaposiciĂ³n]
    Si $\varphi \colon \mathbb{I} \to X$ es un camino de $x$ a $y$, definiendo $\overline{\varphi} \colon \mathbb{I} \to X$ por $\overline{\varphi}(t) = \varphi(1 - t)$, se obtiene un camino de $y$ a $x$. Si, ademĂ¡s, $\psi \colon \mathbb{I} \to X$ es un camino de $y$ a $z$, definiendo $\varphi \cdot \psi \colon \mathbb{I} \to X$ por
    $$\varphi \cdot \psi(t) = \begin{cases} \varphi(2t) & \text{si } 0 \leq t \leq 1/2 \\ \psi(2t - 1) & \text{si } 1/2 \leq t \leq 1 \end{cases}$$
    se obtiene un camino de $x$ a $z$. El camino $\overline{\varphi}$ se denomina \textbf{camino inverso} de $\varphi$ y $\varphi \cdot \psi$ se denomina \textbf{producto} o \textbf{yuxtaposiciĂ³n} de $\varphi$ y $\psi$.
\end{observaciĂ³n}
\begin{definiciĂ³n}[Conexo por caminos]
    Un espacio topolĂ³gico $X$ es \textbf{conexo por caminos} si, para cada par de puntos $x, y \in X$, existe un camino en $X$ de $x$ a $y$. Un subconjunto $Y \subset X$ es conexo por caminos cuando lo es como subespacio.
\end{definiciĂ³n}
\ejemplo{Algunos ejemplos de espacios conexos por caminos: triviales; finitos; convexos y conexos por poligonales en $\R^n$; Kolmogorov.}
\begin{observaciĂ³n}[Arcos y conexidad por arcos]
    Cuando un camino es un mergullo (inmersiĂ³n), se dice que es un \textbf{arco} y se tiene el correspondiente concepto de \textbf{conexidad por arcos}, que en general no es equivalente a la conexidad por caminos, aunque sĂ­ lo es para espacios Hausdorff.
\end{observaciĂ³n}
\begin{proposiciĂ³n}
    Si $\{A_\mu\}_{\mu \in M}$ es una familia de subconjuntos conexos por caminos de un espacio topolĂ³gico $X$ y $\bigcap_{\mu \in M} A_\mu \neq \emptyset$, entonces $\bigcup_{\mu \in M} A_\mu$ es conexo por caminos.
\end{proposiciĂ³n}
\begin{proof}
    Supongamos bajo las condiciones anteriores, que se cumple que $\bigcup_{\mu \in M} A_{\mu}$ no es conexo por caminos, eso quiere decir que existen $x, y \in \bigcup_{\mu \in M} A_{\mu}$ tales que no existe ningĂºn camino de $x$ a $y$. Definamos $\mu_1, \mu_2 \in M$ tales que $x \in A_{\mu_1}$ y $y \in A_{\mu_2}$. Dado que $\bigcap_{\mu \in M} A_\mu \neq \emptyset$, existe $z \in X$ tal que $z \in A_\mu$ para cada $\mu \in M$. Podemos definir los caminos $\varphi_1 \colon \mathbb{I} \to A_{\mu_1}$ y $\varphi_2 \colon \mathbb{I} \to A_{\mu_2}$ como caminos de $x$ a $z$ y de $z$ a $y$, respectivamente. Entonces, la yuxtaposiciĂ³n $\varphi_1 \cdot \varphi_2$ es un camino de $x$ a $y$, lo que contradice la hipĂ³tesis de que no existe ningĂºn camino de $x$ a $y$. Por tanto, $\bigcup_{\mu \in M} A_\mu$ es conexo por caminos.
\end{proof}
\begin{proposiciĂ³n}
    $\bigcup_{\mu \in M} A_\mu$ tambiĂ©n resulta conexo por caminos si se sustituye la condiciĂ³n $\bigcap_{\mu \in M} A_\mu \neq \emptyset$ de la proposiciĂ³n anterior por cualquiera de las siguientes:
    \begin{enumerate}[label=\alph*)]
        \item $A_\mu \cap A_\nu \neq \emptyset$ para todo $\mu, \nu \in M$.
        \item Existe $\nu \in M$ tal que, para todo $\mu \in M$, $A_\mu \cap A_\nu \neq \emptyset$.
        \item $M = \{\mu_i \mid i \in \N\}$ es numerable y, para cada $i \in \N$, $A_{\mu_i} \cap A_{\mu_{i+1}} \neq \emptyset$.
    \end{enumerate}
\end{proposiciĂ³n}
\begin{proof}
    \begin{enumerate}
        \item Sean $x, y \in \bigcup_{\mu \in M} A_\mu$. Entonces existen $\alpha, \beta \in M$ tales que $x \in A_\alpha$ y $y \in A_\beta$. Por hipĂ³tesis, $A_\alpha \cap A_\beta \neq \emptyset$, por lo que existe $z \in X$ tal que $z \in A_\alpha$ y $z \in A_\beta$. Entonces, como $A_\alpha$ es conexo por caminos, existe un camino $\varphi_1 \colon \mathbb{I} \to A_\alpha$ de $x$ a $z$. AnĂ¡logamente, con $A_\beta$ ocurre lo mismo. Por tanto, la yuxtaposiciĂ³n $\varphi_1 \cdot \varphi_2$ es un camino de $x$ a $y$, lo que prueba que $\bigcup_{\mu \in M} A_\mu$ es conexo por caminos.
        \item Es un caso particular del apartado anterior.
        \item Sea $x, y \in \bigcup_{i \in \N} A_{\mu_i}$. Entonces existen $n, m \in \N$ tales que $x \in A_{\mu_n}$ y $y \in A_{\mu_m}$. Supongamos sin pĂ©rdida de generalidad que $n < m$. Por hipĂ³tesis, para cada $i \in \N$ se cumple que $A_{\mu_i} \cap A_{\mu_{i+1}} \neq \emptyset$, por tanto si llamamos $k = m - n$ y definimos $z_i \in A_{\mu_{n+i}} \cap A_{\mu_{n+i+1}}$ para cada $i = 0, \dots, k - 1$, entonces $z_0 \in A_{\mu_n}$, $z_{k-1} \in A_{\mu_m}$ y $z_i \in A_{\mu_{n+i}}$ para cada $i = 0, \dots, k - 1$. Por tanto, como cada $A_{\mu_i}$ es conexo por caminos, existen caminos $\varphi_0 \colon \mathbb{I} \to A_{\mu_n}$ de $x$ a $z_0$, $\varphi_k \colon \mathbb{I} \to A_{\mu_m}$ de $z_{k-1}$ a $y$ y $\varphi_i \colon \mathbb{I} \to A_{\mu_{n+i}}$ de $z_{i-1}$ a $z_i$ para cada $i = 1, \dots, k - 1$. Entonces, la yuxtaposiciĂ³n $\varphi_0 \cdot \varphi_1 \cdot \dots \cdot \varphi_k$ es un camino de $x$ a $y$, lo que prueba que $\bigcup_{i \in \N} A_{\mu_i}$ es conexo por caminos.
    \end{enumerate}
\end{proof}
\begin{proposiciĂ³n}[Imagen continua de un conexo por caminos]
    Si $X$ es conexo por caminos y $f \colon X \to Y$ es sobre y continua, $Y$ es conexo por caminos.
\end{proposiciĂ³n}
\begin{proof}
    Sean $y_1, y_2 \in Y$. Dado que $f$ es sobreyectiva, existen al menos $x_1, x_2 \in X$ tales que $f(x_1) = y_1$ y $f(x_2) = y_2$. Como $X$ es conexo por caminos, existe un camino $\varphi \colon \mathbb{I} \to X$ de $x_1$ a $x_2$. Entonces, la composiciĂ³n $f \circ \varphi \colon \mathbb{I} \to Y$ es un camino de $y_1$ a $y_2$, lo que prueba que $Y$ es conexo por caminos.
\end{proof}
\begin{corolario}
    La conexidad por caminos es una propiedad topolĂ³gica.
\end{corolario}
\begin{proof}
    Por la proposiciĂ³n anterior, si $X$ es conexo por caminos y $f \colon X \to Y$ es continua y sobreyectiva, entonces $Y$ es conexo por caminos. Dado que $f$ es un homeomorfismo, $f^{-1} \colon Y \to X$ tambiĂ©n es continua y sobreyectiva, por lo que si $Y$ es conexo por caminos, entonces $X$ es conexo por caminos.
\end{proof}

\begin{proposiciĂ³n}
    Para subconjuntos de $\R$ y para subconjuntos abiertos de $\R^n$, la conexidad equivale a la conexidad por caminos.
\end{proposiciĂ³n}
\begin{proof}
    La demostraciĂ³n se divide en dos partes.

    \textbf{Parte 1: Subconjuntos de $\R$.} Sea $A \subset \R$ conexo. Veamos que $A$ es conexo por caminos. Primero, observemos que $A$ debe ser un intervalo (posiblemente degenerado).     $$C = \{x \in A \mid \text{existe un camino en } A \text{ de } x_0 \text{ a } x\}.$$
    Veamos que $C$ es abierto: si $x \in C$, como $A$ es abierto, existe $\varepsilon > 0$ tal que $B(x, \varepsilon) \subset A$.     Veamos que $A \setminus C$ es abierto: si $x \in A \setminus C$, como $A$ es abierto, existe $\varepsilon > 0$ tal que $B(x, \varepsilon) \subset A$.     AsĂ­, $C$ y $A \setminus C$ son abiertos en $A$, disjuntos, con $C \cup (A \setminus C) = A$. Como $x_0 \in C$, se tiene $C \neq \emptyset$. Dado que $A$ es conexo, debe ser $A \setminus C = \emptyset$, es decir, $C = A$. Esto prueba que $A$ es conexo por caminos.
\end{proof}

\begin{definiciĂ³n}[RelaciĂ³n de equivalencia]

        \item Transitividad: Si $x \sim y$ y $y \sim z$, entonces existen caminos $\varphi \colon \mathbb{I} \to X$ de $x$ a $y$ y $\psi \colon \mathbb{I} \to X$ de $y$ a $z$. Entonces, la yuxtaposiciĂ³n $\varphi \cdot \psi \colon \mathbb{I} \to X$ es un camino de $x$ a $z$, por lo que $x \sim z$.

\end{proof}
